// Буфер для хранения истории температур (последние 20 значений)
#define MAX_HISTORY 20
float tempHistory[MAX_HISTORY];
unsigned long timeHistory[MAX_HISTORY];
int historyIndex = 0;
bool historyFull = false;

String addrToString(DeviceAddress addr) {
  String str = "";
  for (uint8_t i = 0; i < 8; i++) {
    if (addr[i] < 0x10) str += "0";
    str += String(addr[i], HEX);
    if (i < 7) str += ":";
  }
  return str;
}

// Добавление нового значения в историю
void addTempToHistory(float temp) {
  tempHistory[historyIndex] = temp;
  timeHistory[historyIndex] = millis() / 1000; // секунды с запуска
  historyIndex++;
  if (historyIndex >= MAX_HISTORY) {
    historyIndex = 0;
    historyFull = true;
  }
}

// Генерация JSON-массива для графика
String getChartData() {
  String labels = "[";
  String data = "[";
  int count = historyFull ? MAX_HISTORY : historyIndex;

  // Определяем стартовый индекс для правильного порядка
  int start = historyFull ? historyIndex : 0;
  for (int i = 0; i < count; i++) {
    int idx = (start + i) % MAX_HISTORY;
    if (i > 0) {
      labels += ",";
      data += ",";
    }
    labels += "\"" + String(timeHistory[idx]) + "s\"";
    data += String(tempHistory[idx], 1);
  }
  labels += "]";
  data += "]";
  return "\"labels\": " + labels + ", \"data\": " + data;
}

void handleRoot() {
