/*
 * DS18B20 + RemoteXY — график температуры
 * PlatformIO + ESP32
 * Wi-Fi: GalaxyM / www111www
 * Подключение: GPIO4, с подтяжкой 4.7 кОм
 */

#define REMOTEXY_MODE__WIFI_POINT
#include <WiFi.h>

#define REMOTEXY_WIFI_SSID "GalaxyM"
#define REMOTEXY_WIFI_PASSWORD "www111www"
#define REMOTEXY_SERVER_PORT 6377

#include <RemoteXY.h>
#include <OneWire.h>
#include <DallasTemperature.h>

// Конфигурация интерфейса RemoteXY (Online Graph)
#pragma pack(push, 1)
uint8_t RemoteXY_CONF[] = {
  255,0,0,4,0,21,0,19,0,0,0,0,31,1,106,200,1,1,1,0,
  68,1,2,105,72,1,8,36
};
struct {
  float onlineGraph_01_var1; // выходная переменная для графика
  uint8_t connect_flag;
} RemoteXY;
#pragma pack(pop)

#define DS18B20_PIN 4
OneWire oneWire(DS18B20_PIN);
DallasTemperature sensors(&oneWire);

void setup() {
  sensors.begin();
  RemoteXY_Init();
}

void loop() {
  RemoteXY_Handler();

  sensors.requestTemperatures();
  float t = sensors.getTempCByIndex(0);
  if (t == DEVICE_DISCONNECTED_C) {
    t = -99.9f;
  }

  RemoteXY.onlineGraph_01_var1 = t;

  // Обновление каждые 2 секунды (без delay!)
  static uint32_t lastUpdate = 0;
  if (millis() - lastUpdate >= 2000) {
    lastUpdate = millis();
    // RemoteXY_Handler() уже вызван — данные уйдут автоматически
  }
}
